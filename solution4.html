Код javascript:

            
  const companies = [
    { id: 1, name: "Hexlet", description: "online courses" },
    { id: 2, name: "Google", description: "search engine" },
    { id: 3, name: "Facebook", description: "social network" },
  ];
  const uiState = {
    selectedCompanyId: null,
  };
  const render = (companies, buttonsContainer) => {
    const buttons = companies.map(({ id, name }) => {
      const button = document.createElement("button");
      button.classList.add("btn", "btn-primary", "mr-3");
      button.style.marginRight = '10px';
      button.textContent = name;
      button.addEventListener("click", (e) => {
        e.preventDefault();
        if (uiState.selectedCompanyId === id) {
          uiState.selectedCompanyId = null;
        } else {
          uiState.selectedCompanyId = id;
        }
        render(companies, buttonsContainer);
      });

      return button;
    });

    if (uiState.selectedCompanyId) {
      const company = companies.find(
        (item) => item.id === uiState.selectedCompanyId
      );
      if (company) {
        const div = document.createElement("div");
        div.textContent = company.description;
        buttons.push(div);
      }
    }
    buttonsContainer.replaceChildren(...buttons);
  };
  const app = (companies) => {
    const buttonsContainer = document.querySelector(".company-container");
    render(companies, buttonsContainer);
  };
  app(companies);
